# PUT - додати картинку у кеш
curl -X PUT --data-binary "@200.jpeg" http://127.0.0.1:3001/200 -v

# GET - отримати картинку з кешу
curl -X GET http://127.0.0.1:3001/200 --output downloaded_200.jpeg -v

# DELETE - видалити картинку з кешу
curl -X DELETE http://127.0.0.1:3001/200 -v

# GET - картинка не існує (повинен бути 404)
curl -X GET http://127.0.0.1:3001/999 -v

# POST - заборонений метод (повинен бути 405)
curl -X POST http://127.0.0.1:3001/200 -v
